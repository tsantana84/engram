# Tick Log Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add per-tick sync logging to the local engram worker, stored in SQLite and viewable at `localhost:37777/ticks` in the same brutalist design as the learnings approval page.

**Architecture:** SyncWorker instruments its `tick()` method with a try/finally block that writes a `TickRecord` to a new `tick_log` SQLite table (migration 33) via `SessionStore`. A new `GET /api/ticks` endpoint in `ViewerRoutes` reads recent ticks. A standalone `plugin/ui/ticks.html` page (pure HTML/CSS/JS, no build step) renders the data using safe DOM APIs.

**Tech Stack:** TypeScript/Bun, bun:sqlite, Express (existing worker server), vanilla JS + inline CSS for the UI page.

```ts
/**
 * Create tick_log table (migration 33)
 */
private createTickLogTable(): void {
  const applied = this.db
    .prepare('SELECT version FROM schema_versions WHERE version = ?')
    .get(33) as SchemaVersion | undefined;
  if (applied) return;

  this.db.run(`
    CREATE TABLE IF NOT EXISTS tick_log (
      id                  INTEGER PRIMARY KEY AUTOINCREMENT,
      ticked_at           INTEGER NOT NULL DEFAULT (unixepoch()),
      agent_name          TEXT    NOT NULL DEFAULT '',
      duration_ms         INTEGER NOT NULL,
      sessions_extracted  INTEGER NOT NULL DEFAULT 0,
      learnings_enqueued  INTEGER NOT NULL DEFAULT 0,
      items_pushed        INTEGER NOT NULL DEFAULT 0,
      items_failed        INTEGER NOT NULL DEFAULT 0,
      queue_depth_after   INTEGER NOT NULL DEFAULT 0,
      errors              TEXT
    )
  `);
  this.db
    .prepare('INSERT OR IGNORE INTO schema_versions (version, applied_at) VALUES (?, ?)')
    .run(33, new Date().toISOString());
  logger.debug('DB', 'Migration 33 applied: tick_log table created');
}

insertTickLog(record: TickRecord): void {
  this.db.prepare(`
    INSERT INTO tick_log
      (agent_name, duration_ms, sessions_extracted, learnings_enqueued,
       items_pushed, items_failed, queue_depth_after, errors)
    VALUES (?, ?, ?, ?, ?, ?, ?, ?)
  `).run(
    record.agent_name,
    record.duration_ms,
    record.sessions_extracted,
    record.learnings_enqueued,
    record.items_pushed,
    record.items_failed,
    record.queue_depth_after,
    record.errors.length > 0 ? JSON.stringify(record.errors) : null,
  );
  // Retention: keep last 1000 rows
  this.db.run(
    `DELETE FROM tick_log WHERE id < (SELECT MAX(id) - 999 FROM tick_log)`
  );
}

getTickLog(limit: number): Array<TickRecord & { id: number; ticked_at: number }> {
  const rows = this.db.prepare(`
    SELECT id, ticked_at, agent_name, duration_ms, sessions_extracted,
           learnings_enqueued, items_pushed, items_failed, queue_depth_after, errors
    FROM tick_log ORDER BY ticked_at DESC LIMIT ?
  `).all(limit) as Array<any>;

  return rows.map((r) => ({
    id: r.id,
    ticked_at: r.ticked_at,
    agent_name: r.agent_name,
    duration_ms: r.duration_ms,
    sessions_extracted: r.sessions_extracted,
    learnings_enqueued: r.learnings_enqueued,
    items_pushed: r.items_pushed,
    items_failed: r.items_failed,
    queue_depth_after: r.queue_depth_after,
    errors: r.errors ? JSON.parse(r.errors) : [],
  }));
}
```

```ts
async tick(): Promise<void> {
  if (!this.enabled || this.paused) return;

  const startMs = Date.now();
  const record = {
    agent_name: this.agentName,
    duration_ms: 0,
    sessions_extracted: 0,
    learnings_enqueued: 0,
    items_pushed: 0,
    items_failed: 0,
    queue_depth_after: 0,
    errors: [] as string[],
  };

  try {
    // 1. Extract pending/retryable sessions into queued learnings.
    if (this.extractionEnabled && this.extractor) {
      const sessions = this.sessionStore.getPendingExtractionSessions(5);
      for (const s of sessions) {
        try {
          const enqueued = await this.extractSessionLearnings(s.id);
          record.sessions_extracted++;
          record.learnings_enqueued += enqueued;
        } catch { /* extractSessionLearnings handles its own errors */ }
      }
    }

    // 2. Drain the sync queue.
    const pending = this.queue.getPending(this.batchSize);

    const learningItems = pending.filter((p) => p.entity_type === 'learning');
    const legacyItems = pending.filter((p) => p.entity_type !== 'learning');

    const approvedItems = learningItems.filter((p) => p.target_status === 'approved');
    const pendingItems = learningItems.filter((p) => p.target_status === 'pending');

    for (const [group, status] of [
      [approvedItems, 'approved'] as const,
      [pendingItems, 'pending'] as const,
    ]) {
      if (group.length === 0) continue;
      try {
        await this.client.pushLearnings(
          group.map((g) => g.payload!).filter(Boolean),
          status
        );
        this.queue.markSynced(group.map((g) => g.id));
        record.items_pushed += group.length;
      } catch (err: any) {
        this.handlePushError(err, group.map((g) => g.id));
        record.items_failed += group.length;
        record.errors.push(err?.message ?? 'unknown error');
      }
    }

    // Legacy observation/session/summary path.
    if (!this.extractionEnabled && legacyItems.length > 0) {
      const payload = this.buildPayload(legacyItems);
      payload.observations = await this.processConflicts(payload.observations);
      const ids = legacyItems.map((i) => i.id);
      try {
        await this.client.push(payload);
        this.queue.markSynced(ids);
        record.items_pushed += ids.length;
      } catch (err: any) {
        this.handlePushError(err, ids);
        record.items_failed += ids.length;
        record.errors.push(err?.message ?? 'unknown error');
      }
    } else if (legacyItems.length > 0) {
      this.queue.markSynced(legacyItems.map((i) => i.id));
    }

    record.queue_depth_after = this.queue.countPending();
  } finally {
    record.duration_ms = Date.now() - startMs;
    this.sessionStore.insertTickLog(record);
  }
}
```

```ts
private handleGetTicks = this.wrapHandler((req: Request, res: Response): void => {
  const limitParam = parseInt(String(req.query.limit ?? '100'), 10);
  const limit = Number.isFinite(limitParam) ? Math.min(limitParam, 500) : 100;
  const store = this.dbManager.getSessionStore();
  const ticks = store.getTickLog(limit);
  res.json({ ticks, fetchedAt: new Date().toISOString() });
});

private handleTicksUI = this.wrapHandler((req: Request, res: Response): void => {
  const packageRoot = getPackageRoot();
  const candidates = [
    path.join(packageRoot, 'ui', 'ticks.html'),
    path.join(packageRoot, 'plugin', 'ui', 'ticks.html'),
  ];
  const ticksPath = candidates.find(p => existsSync(p));
  if (!ticksPath) {
    throw new Error('Ticks UI not found — run npm run build-and-sync');
  }
  const html = readFileSync(ticksPath, 'utf-8');
  res.setHeader('Content-Type', 'text/html');
  res.send(html);
});
```

```html
<script>
(function () {
  'use strict';
  var allTicks = [], activeAgent = 'all', timer = null;

  function esc(s) { return String(s ?? ''); }

  function formatTime(epoch) {
    var d = new Date(epoch * 1000);
    return d.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit', second: '2-digit', hour12: false });
  }

  function formatDur(ms) {
    return ms < 1000 ? ms + 'ms' : (ms / 1000).toFixed(1) + 's';
  }

  function status(t) {
    if (t.items_pushed === 0 && t.items_failed === 0 && t.sessions_extracted === 0) return 'idle';
    if (t.items_failed === 0) return 'ok';
    if (t.items_pushed > 0) return 'partial';
    return 'fail';
  }

  function makeTd(text, cls) {
    var td = document.createElement('td');
    td.textContent = esc(text);
    if (cls) td.className = cls;
    return td;
  }

  function makeNumTd(n) {
    var td = document.createElement('td');
    td.textContent = String(n);
    if (n === 0) td.className = 'zero';
    else if (n < 0) td.className = 'err-count';
    return td;
  }

  function renderTable(ticks) {
    var wrap = document.getElementById('table-wrap');
    while (wrap.firstChild) wrap.removeChild(wrap.firstChild);

    if (ticks.length === 0) {
      var empty = document.createElement('div');
      empty.className = 'empty';
      empty.textContent = 'No ticks recorded yet. Ticks appear after the worker runs its first cycle.';
      wrap.appendChild(empty);
      return;
    }

    var table = document.createElement('table');
    var thead = document.createElement('thead');
    var hrow = document.createElement('tr');
    ['Time','Agent','Status','Duration','Extracted','Enqueued','Pushed','Failed','Queue'].forEach(function(h) {
      var th = document.createElement('th');
      th.textContent = h;
      hrow.appendChild(th);
    });
    thead.appendChild(hrow);
    table.appendChild(thead);

    var tbody = document.createElement('tbody');
    ticks.forEach(function (t) {
      var s = status(t);
      var tr = document.createElement('tr');
      if (s !== 'idle') tr.className = 'row-' + s;

      // Time
      var tdTime = document.createElement('td');
      tdTime.className = 'ts';
      tdTime.textContent = formatTime(t.ticked_at);
      tr.appendChild(tdTime);

      // Agent
      var tdAgent = document.createElement('td');
      var tag = document.createElement('span');
      tag.className = 'agent-tag';
      tag.textContent = esc(t.agent_name) || '—';
      tdAgent.appendChild(tag);
      tr.appendChild(tdAgent);

      // Status badge
      var tdStatus = document.createElement('td');
      var badge = document.createElement('span');
      badge.className = 'badge badge-' + s;
      badge.textContent = s.toUpperCase();
      tdStatus.appendChild(badge);
      tr.appendChild(tdStatus);

      // Duration
      var tdDur = document.createElement('td');
      tdDur.className = 'dur';
      tdDur.textContent = formatDur(t.duration_ms);
      tr.appendChild(tdDur);

      tr.appendChild(makeNumTd(t.sessions_extracted));
      tr.appendChild(makeNumTd(t.learnings_enqueued));
      tr.appendChild(makeNumTd(t.items_pushed));

      // Failed (red if > 0)
      var tdFail = document.createElement('td');
      tdFail.textContent = String(t.items_failed);
      if (t.items_failed > 0) tdFail.className = 'err-count';
      else tdFail.className = 'zero';
      tr.appendChild(tdFail);

      tr.appendChild(makeNumTd(t.queue_depth_after));
      tbody.appendChild(tr);

      // Error detail rows
      if (t.errors && t.errors.length > 0) {
        var errTr = document.createElement('tr');
        errTr.className = 'error-row';
        var errTd = document.createElement('td');
        errTd.colSpan = 9;
        t.errors.forEach(function(e, i) {
          if (i > 0) errTd.appendChild(document.createElement('br'));
          errTd.appendChild(document.createTextNode('✗ ' + esc(e)));
        });
        errTr.appendChild(errTd);
        tbody.appendChild(errTr);
      }
    });

    table.appendChild(tbody);
    wrap.appendChild(table);
  }

  function buildAgentBar(ticks) {
    var agents = [];
    ticks.forEach(function(t) { if (t.agent_name && agents.indexOf(t.agent_name) === -1) agents.push(t.agent_name); });
    agents.sort();
    var bar = document.getElementById('agent-bar');
    while (bar.firstChild) bar.removeChild(bar.firstChild);

    var allBtn = document.createElement('div');
    allBtn.className = 'agent-btn' + (activeAgent === 'all' ? ' active' : '');
    allBtn.dataset.agent = 'all';
    allBtn.textContent = 'All';
    bar.appendChild(allBtn);

    agents.forEach(function(name) {
      var btn = document.createElement('div');
      btn.className = 'agent-btn' + (activeAgent === name ? ' active' : '');
      btn.dataset.agent = name;
      btn.textContent = name;
      bar.appendChild(btn);
    });

    bar.querySelectorAll('.agent-btn').forEach(function(btn) {
      btn.addEventListener('click', function() {
        activeAgent = btn.dataset.agent;
        bar.querySelectorAll('.agent-btn').forEach(function(b) { b.classList.remove('active'); });
        btn.classList.add('active');
        filterAndRender();
      });
    });
  }

  function filterAndRender() {
    var visible = activeAgent === 'all' ? allTicks : allTicks.filter(function(t) { return t.agent_name === activeAgent; });
    renderTable(visible);
  }

  function load() {
    fetch('/api/ticks?limit=100')
      .then(function(r) { if (!r.ok) throw new Error('HTTP ' + r.status); return r.json(); })
      .then(function(data) {
        allTicks = data.ticks || [];
        var fa = document.getElementById('fetched-at');
        fa.textContent = 'Fetched ' + new Date(data.fetchedAt).toLocaleTimeString();
        buildAgentBar(allTicks);
        filterAndRender();
      })
      .catch(function(err) {
        var wrap = document.getElementById('table-wrap');
        while (wrap.firstChild) wrap.removeChild(wrap.firstChild);
        var div = document.createElement('div');
        div.className = 'empty';
        div.style.color = 'var(--red)';
        div.textContent = 'Failed to load: ' + err.message;
        wrap.appendChild(div);
      });
  }

  function scheduleRefresh() {
    if (timer) clearInterval(timer);
    timer = setInterval(load, 60000);
  }

  document.getElementById('btn-refresh').addEventListener('click', function() {
    load(); scheduleRefresh();
  });

  load();
  scheduleRefresh();
})();
</script>
</body>
</html>
```
